```tsx
// import { Overlay, Element } from '@vitus-labs/elements'

const Button = (props) => {
  const { active } = props

  return (
    <Element
      {...props}
      label="Click on me"
      tag="button"
      style={{
        background: active ? '#54D6FF' : 'rgba(84, 214, 255, 0.4)',
        border: '1px solid #54D6FF',
        padding: '8px 16px',
        color: '#fff',
        borderRadius: 4,
        outline: 'none',
      }}
    />
  )
}

const Content = (props) => (
  <Element
    style={{
      backgroundColor: '#fff',
      color: '#000',
      padding: 8,
      borderRadius: 8,
    }}
    {...props}
  >
    Some content here
  </Element>
)

// ↓↓↓ Overlay ↓↓↓
render(
  <Overlay trigger={Button} refName="innerRef" offsetY={8}>
    <Content />
  </Overlay>
)
```
