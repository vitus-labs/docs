```tsx
// import { Overlay, Element } from '@vitus-labs/elements'

const Button = (props) => (
  <Element
    {...props}
    label="Hover on me"
    tag="button"
    style={{
      background: 'rgba(84, 214, 255, 0.4)',
      border: '1px solid #54D6FF',
      padding: '8px 16px',
      color: '#fff',
      borderRadius: 4,
      outline: 'none',
    }}
  />
)

const Content = (props) => (
  <Element
    style={{
      backgroundColor: '#fff',
      color: '#000',
      padding: 8,
      borderRadius: 8,
    }}
    {...props}
  >
    Some content here
  </Element>
)

// ↓↓↓ Overlay ↓↓↓
render(
  <Overlay
    trigger={Button}
    refName="innerRef"
    openOn="hover"
    closeOn="hover"
    align="top"
    alignX="center"
    offsetY={8}
  >
    <Content />
  </Overlay>
)
```
